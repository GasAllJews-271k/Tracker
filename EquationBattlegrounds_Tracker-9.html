Step 1 — DataStore System
Permanent player data. Coins, progress, unlocks — all saved here.
▾
Why this step now? Every other system reads and writes player data. If you skip this, nothing persists between sessions. Build it first, test it in isolation, then every future system just calls DataManager.Get(player).

Enable DataStore in Studio
⚙️ Studio Settings
  • Top bar → Home → Game Settings → Security tab
  • Toggle ON: Enable Studio Access to API Services
  • This only affects Studio playtests. Published games have access automatically.
  • If you skip this, DataStore silently returns nil for every load and all data looks like it's saving but nothing is actually stored.

-- PART 1: SERVICE SETUP & PLAYER DEFAULTS
-- ════════════════════════════════════════

local DataStoreService = game:GetService("DataStoreService")
local Players          = game:GetService("Players")
local RunService       = game:GetService("RunService")

local DataManager = {}

-- The DataStore name. NEVER change this after players have real data.
-- Changing it is like creating a brand new empty database.
local PlayerStore = DataStoreService:GetDataStore("EQBattlegrounds_v1")

-- Server-side memory cache: { [userId] = dataTable }
-- This is what every server script reads. Never expose to clients.
local cache     = {}
local lastSaved = {}   -- { [userId] = tick() } for auto-save throttle

-- ── DEFAULT DATA ─────────────────────────────────────────────────
-- This is what every brand new player starts with.
-- Rule: NEVER remove a field once the game is live.
--       ADD new fields with sensible defaults instead.
--       The mergeData() function below handles old saves getting new fields.
local DEFAULTS = {
    -- Economy
    coins            = 50,       -- starting coins

    -- Combat record
    wins             = 0,
    losses           = 0,
    totalBattles     = 0,
    longestWinStreak = 0,
    currentWinStreak = 0,

    -- Progression
    ownedVolumes     = {},       -- { 1, 2, 3 } — which BAPGTV volumes purchased
    completedQuizzes = {},       -- { "vol1_sec1", "vol2_sec3" }
    unlockedEqs      = {},       -- equation IDs the player can use in battle

    -- Active spin state (persists between sessions)
    spinSubject      = nil,
    spinSubSubject   = nil,
    spinTier         = nil,

    -- Cosmetics
    selectedBorder   = "default",
    ownedBorders     = { "default" },

    -- Meta
    joinDate         = os.time(),
    lastLoginDate    = "",
    totalPlayTime    = 0,        -- seconds (updated on save)
    sessionStart     = 0,        -- set on load, used to calculate play time
}

-- Deep clone so each player gets their own independent table
local function deepCopy(t)
    local copy = {}
    for k, v in pairs(t) do
        copy[k] = type(v) == "table" and deepCopy(v) or v
    end
    return copy
end

-- Merge saved data onto fresh defaults.
-- This means if you ADD a new field to DEFAULTS later,
-- existing players automatically get it next login.
local function mergeDefaults(saved)
    local base = deepCopy(DEFAULTS)
    if type(saved) ~= "table" then return base end
    for k, v in pairs(saved) do
        base[k] = v
    end
    return base
end

-- PART 2: LOADING DATA

-- Retry wrapper — DataStores occasionally fail with temporary errors.
-- We retry up to 3 times with exponential back-off before giving up.
local function tryGet(key)
    local lastErr
    for attempt = 1, 3 do
        local ok, result = pcall(function()
            return PlayerStore:GetAsync(key)
        end)
        if ok then
            return true, result
        end
        lastErr = result
        warn(string.format("[DataManager] GetAsync attempt %d failed: %s", attempt, tostring(result)))
        if attempt < 3 then task.wait(2 ^ attempt) end  -- 2s, 4s back-off
    end
    return false, lastErr
end

function DataManager.Load(player)
    local key    = tostring(player.UserId)
    local ok, raw = tryGet(key)

    local data
    if ok then
        data = mergeDefaults(raw)
        if raw == nil then
            print("[DataManager] New player:", player.Name)
        else
            print("[DataManager] Loaded:", player.Name, "| Coins:", data.coins)
        end
    else
        -- All retries failed — use defaults so player can still play
        -- but flag the session as potentially unsafe to save
        warn("[DataManager] LOAD FAILED for", player.Name, "— using temporary defaults")
        data = deepCopy(DEFAULTS)
        data._loadFailed = true
    end

    -- Record session start time for play time tracking
    data.sessionStart = os.time()
    data.lastLoginDate = os.date("%Y-%m-%d")

    cache[player.UserId] = data
    return data
end

-- Read cached data for any server script
function DataManager.Get(player)
    return cache[player.UserId]
end

-- Set a single field in a player's cached data
function DataManager.Set(player, key, value)
    local d = cache[player.UserId]
    if d then d[key] = value end
end

-- PART 3: SAVING DATA
-- ════════════════════════════════════════

local AUTOSAVE_EVERY = 300   -- seconds (5 minutes)

local function trySave(key, data)
    local lastErr
    for attempt = 1, 3 do
        local ok, err = pcall(function()
            PlayerStore:SetAsync(key, data)
        end)
        if ok then return true end
        lastErr = err
        warn(string.format("[DataManager] SetAsync attempt %d failed: %s", attempt, tostring(err)))
        if attempt < 3 then task.wait(2 ^ attempt) end
    end
    return false, lastErr
end

function DataManager.Save(player)
    local d = cache[player.UserId]
    if not d then return end

    -- Don't save a session that loaded with bad data — we'd overwrite good data with defaults
    if d._loadFailed then
        warn("[DataManager] Refusing to save", player.Name, "— data was not loaded cleanly")
        return
    end

    -- Update total play time before saving
    if d.sessionStart and d.sessionStart > 0 then
        d.totalPlayTime = (d.totalPlayTime or 0) + (os.time() - d.sessionStart)
        d.sessionStart  = os.time()   -- reset for next save interval
    end

    -- Strip internal-only flags before saving to DataStore
    local toSave = deepCopy(d)
    toSave._loadFailed  = nil
    toSave.sessionStart = nil

    local key = tostring(player.UserId)
    local ok, err = trySave(key, toSave)

    if ok then
        lastSaved[player.UserId] = tick()
        print("[DataManager] Saved:", player.Name)
    else
        warn("[DataManager] ALL SAVE RETRIES FAILED for:", player.Name, err)
    end
end

-- Auto-save loop — checks every 30s, saves if interval has passed
task.spawn(function()
    while true do
        task.wait(30)
        local now = tick()
        for _, player in ipairs(Players:GetPlayers()) do
            local uid  = player.UserId
            local last = lastSaved[uid] or 0
            if cache[uid] and (now - last) >= AUTOSAVE_EVERY then
                task.spawn(DataManager.Save, player)
            end
        end
    end
end)

-- Save on leave
Players.PlayerRemoving:Connect(function(player)
    DataManager.Save(player)
    task.wait(0.2)   -- give SetAsync a moment before the data is wiped
    cache[player.UserId]     = nil
    lastSaved[player.UserId] = nil
end)

-- Save everyone on server shutdown
-- Roblox gives ~30 seconds before force-closing
game:BindToClose(function()
    print("[DataManager] Server shutting down — saving all players...")
    for _, player in ipairs(Players:GetPlayers()) do
        task.spawn(DataManager.Save, player)
    end
    task.wait(4)
    print("[DataManager] Shutdown save complete.")
end)

-- Auto-connect load on join
Players.PlayerAdded:Connect(DataManager.Load)

-- PART 4: COIN TRANSACTIONS

-- All coin math happens here, server-side only.
-- The client never decides its own coin balance.

local function pushCoinUpdate(player, message)
    -- Fires the CoinUpdate event so the client HUD refreshes
    -- We get Events lazily so DataManager doesn't need to require EventSetup
    local Events = game:GetService("ReplicatedStorage"):FindFirstChild("Events")
    if Events and Events:FindFirstChild("CoinUpdate") then
        local d = cache[player.UserId]
        Events.CoinUpdate:FireClient(player, d and d.coins or 0, message)
    end
end

-- Award coins to a player
-- Returns: true on success, false if player has no data
function DataManager.AddCoins(player, amount, message)
    local d = cache[player.UserId]
    if not d then return false end
    if type(amount) ~= "number" or amount < 0 then return false end
    d.coins = (d.coins or 0) + math.floor(amount)
    pushCoinUpdate(player, message)
    return true
end

-- Spend coins — validates balance first
-- Returns: true, nil  on success
--          false, "reason" on failure
function DataManager.SpendCoins(player, amount)
    local d = cache[player.UserId]
    if not d then
        return false, "Player data not loaded"
    end
    if type(amount) ~= "number" or amount <= 0 then
        return false, "Invalid amount"
    end
    local bal = d.coins or 0
    if bal < amount then
        return false, string.format("Not enough coins (need %d, have %d)", amount, bal)
    end
    d.coins = bal - math.floor(amount)
    pushCoinUpdate(player, nil)
    return true, nil
end

-- Simple balance read (server-side only)
function DataManager.GetCoins(player)
    local d = cache[player.UserId]
    return d and d.coins or 0
end

return DataManager

Understand the coin flow
ℹ️ Good to Know
  • The server is the only source of truth for coins. Here's the full flow:
  • Earn: Server calls DataManager.AddCoins(player, 25, '+25 coins!') → updates cache → fires CoinUpdate event
  • Spend: Server calls DataManager.SpendCoins(player, 50) → checks balance → deducts → fires CoinUpdate

-- StarterPlayerScripts > HudController
-- Updates the coin HUD whenever the server changes the player's balance

local Players      = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local RS           = game:GetService("ReplicatedStorage")

local player   = Players.LocalPlayer
local playerGui= player:WaitForChild("PlayerGui")

-- Wait for GUI elements (they load async)
local hudGui    = playerGui:WaitForChild("HudGui")
local coinLabel = hudGui:WaitForChild("CoinFrame"):WaitForChild("CoinLabel")
local coinToast = hudGui:WaitForChild("CoinToast")

-- Wait for the Events folder (created by EventSetup on the server)
local Events = RS:WaitForChild("Events")

-- Track displayed value for smooth counting animation
local displayedCoins = 50
local countTween

-- Smoothly count the coin number up or down over 0.4 seconds
local function animateCoinCount(targetCoins)
    if countTween then countTween:Cancel() end

    local startVal = displayedCoins
    local diff     = targetCoins - startVal
    local duration = math.clamp(math.abs(diff) / 200, 0.15, 0.5)
    local startTime= tick()

    countTween = TweenService:Create(
        Instance.new("NumberValue"),  -- dummy object for TweenService
        TweenInfo.new(duration, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
        {}
    )

    -- Manual lerp via Heartbeat instead (TweenService doesn't lerp numbers easily)
    local conn
    conn = game:GetService("RunService").Heartbeat:Connect(function()
        local t = math.clamp((tick() - startTime) / duration, 0, 1)
        local eased = t < 0.5 and 2*t*t or -1+(4-2*t)*t   -- ease in-out quad
        local current = math.round(startVal + diff * eased)
        coinLabel.Text  = "🪙 " .. tostring(current)
        displayedCoins  = current
        if t >= 1 then conn:Disconnect() end
    end)
end

-- Show a toast message then fade it out
local toastTween
local function showToast(message, isPositive)
    if toastTween then toastTween:Cancel() end

    coinToast.Text             = message
    coinToast.TextColor3       = isPositive
        and Color3.fromRGB(80, 220, 100)   -- green for gains
        or  Color3.fromRGB(220, 80,  80)   -- red for spending
    coinToast.TextTransparency = 0
    coinToast.Visible          = true

    -- Hold for 1.8 seconds then fade out
    task.delay(1.8, function()
        toastTween = TweenService:Create(coinToast,
            TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
            { TextTransparency = 1 }
        )
        toastTween:Play()
        toastTween.Completed:Once(function()
            coinToast.Visible = false
        end)
    end)
end

-- ── Listen for coin updates from the server ───────────────────────
Events:WaitForChild("CoinUpdate").OnClientEvent:Connect(function(newBalance, message)
    -- Determine if coins went up or down for toast colour
    local gained = newBalance > displayedCoins

    animateCoinCount(newBalance)

    if message and message ~= "" then
        showToast(message, gained)
    end

    -- Pulse the coin label gold → white → gold on change
    TweenService:Create(coinLabel,
        TweenInfo.new(0.12, Enum.EasingStyle.Quad),
        { TextColor3 = Color3.fromRGB(255, 255, 255) }
    ):Play()
    task.delay(0.12, function()
        TweenService:Create(coinLabel,
            TweenInfo.new(0.2, Enum.EasingStyle.Quad),
            { TextColor3 = Color3.fromRGB(255, 200, 60) }
        ):Play()
    end)
end)

print("[HudController] Ready")

Common mistakes to avoid
⚠️ Common Mistake
  • Don't use :FindFirstChild in LocalScripts for GUI — use :WaitForChild or the GUI won't exist yet when the script runs
  • Don't let the client fire a 'SetCoins' event — always send the new balance from the server after a transaction

  • Do call DataManager.AddCoins for every coin reward — it handles the HUD update automatically

Step 3 — RemoteEvent System
The communication backbone. Every system uses this.
▾
Why this step now? RemoteEvents are how your GUIs (client) talk to your game logic (server). Without this nothing interactive works — the spin button can't request a spin, the equation input can't submit, challenges can't be sent. Set it up once and all future scripts just use the event names.
1
Understand client-server communication
ℹ️ Good to Know
  • Roblox separates code into Server (Scripts) and Client (LocalScripts). They can't call each other's functions directly.
  • Client → Server: Player clicks a button → LocalScript fires Event:FireServer(data) → Server handler receives it
  • Server → Client: Server finishes processing → fires Event:FireClient(player, data) → Client updates the UI
  • RemoteEvents live in ReplicatedStorage because both server and client can see that service.
  • Golden rule: Validate all data on the server. Assume any client can send any data at any time.
2
Create EventSetup script
🏗️ Explorer Setup
  • ServerScriptService → Insert Object → Script → rename EventSetup
  • This must be a Script (not ModuleScript) so it runs automatically when the server starts
  • It runs before any LocalScript, so clients can safely WaitForChild() for the events

-- ServerScriptService > EventSetup
-- Creates all RemoteEvents, rate limiting, and server handlers
-- ═══════════════════════════════════════════════════════════════

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players           = game:GetService("Players")

-- Lazy-require modules so they load after this script's top runs
local function getDataManager()
    return require(script.Parent:WaitForChild("DataManager"))
end
local function getSpinSystem()
    return require(script.Parent:WaitForChild("SpinSystem"))
end
local function getBattleManager()
    return require(script.Parent:WaitForChild("BattleManager"))
end

-- ── Create Events folder ──────────────────────────────────────────
local Events = Instance.new("Folder")
Events.Name   = "Events"
Events.Parent = ReplicatedStorage

local function E(name)   -- RemoteEvent shorthand
    local e = Instance.new("RemoteEvent")
    e.Name   = name
    e.Parent = Events
    return e
end

-- Economy
local eCoinUpdate     = E("CoinUpdate")       -- S→C: new balance + message
local eSpinRequest    = E("SpinRequest")       -- C→S: player wants to spin
local eSpinResult     = E("SpinResult")        -- S→C: spin outcome
local ePurchaseVolume = E("PurchaseVolume")    -- C→S: buy a book volume
local eShopMessage    = E("ShopMessage")       -- S→C: shop feedback

-- Battle
local eSubmitEquation  = E("SubmitEquation")   -- C→S: player types equation
local eBattleUpdate    = E("BattleUpdate")     -- S→C: state changes, cards, health
local eChallengeOut    = E("ChallengeOut")     -- C→S: send a challenge
local eChallengeIn     = E("ChallengeIn")      -- S→C: incoming challenge
local eChallengeAck    = E("ChallengeAck")     -- C→S: accept/decline

-- Admin
local eAdminCommand    = E("AdminCommand")     -- C→S: admin command
local eAdminResult     = E("AdminResult")      -- S→C: command result

print("[EventSetup] Created", #Events:GetChildren(), "RemoteEvents")

-- ── Rate limiting ─────────────────────────────────────────────────
-- Prevents exploiters from spamming events thousands of times per second
local rateLimits = {}

local RATE = {
    SpinRequest    = 2.0,
    SubmitEquation = 0.4,
    ChallengeOut   = 5.0,
    ChallengeAck   = 1.0,
    PurchaseVolume = 1.5,
    AdminCommand   = 0.3,
}

local function checkRate(player, eventName)
    local uid   = player.UserId
    local limit = RATE[eventName]
    if not limit then return true end

    rateLimits[uid] = rateLimits[uid] or {}
    local last = rateLimits[uid][eventName] or 0
    local now  = tick()

    if now - last < limit then
        warn(string.format("[RateLimit] %s fired '%s' too fast (%.2fs gap, limit %.2fs)",
            player.Name, eventName, now - last, limit))
        return false
    end
    rateLimits[uid][eventName] = now
    return true
end

Players.PlayerRemoving:Connect(function(p)
    rateLimits[p.UserId] = nil
end)

-- ── SERVER HANDLERS ───────────────────────────────────────────────

-- Spin
eSpinRequest.OnServerEvent:Connect(function(player)
    if not checkRate(player, "SpinRequest") then return end

    local DataManager = getDataManager()

-- ── Core: weighted random picker ──────────────────────────────────
function SpinSystem.weightedRandom(pool)
    local total = 0
    for _, item in ipairs(pool) do total += item.weight end

    local roll = math.random(1, total)
    local cum  = 0
    for _, item in ipairs(pool) do
        cum += item.weight
        if roll <= cum then return item end
    end
    return pool[#pool]  -- fallback (shouldn't reach here)
end

-- ── Full 3-stage spin ─────────────────────────────────────────────
function SpinSystem.Spin()
    local subject    = SpinSystem.weightedRandom(SUBJECTS)
    local subList    = SUB_SUBJECTS[subject.name] or { "General" }
    local subSubject = subList[math.random(#subList)]
    local tier       = SpinSystem.weightedRandom(TIERS)

    return {
        subject      = subject.name,
        subjectColor = subject.color,
        subSubject   = subSubject,
        tier         = tier.name,
        tierMult     = tier.mult,
        tierColor    = tier.color,
    }
end

-- Matches player input to equations using exact match first,
-- then Levenshtein distance for closest-match display on wrong answers.

local EquationDatabase = require(script.Parent.EquationDatabase)

local EquationEngine   = {}

-- ── String normalisation ──────────────────────────────────────────
-- Applied to both player input AND stored formats before comparing.
-- This means "F = MA", "f=ma", and "f = ma" all match the same entry.
local function normalize(s)
    s = tostring(s):lower()        -- lowercase everything
    s = s:gsub("%s+", " ")         -- collapse multiple spaces to one
    s = s:match("^%s*(.-)%s*$")    -- trim leading/trailing whitespace
    return s
end

-- Version with all spaces removed — used for a second looser pass
local function compact(s)
    return normalize(s):gsub(" ", "")
end

EquationEngine.normalize = normalize  -- expose for other modules

-- ── Levenshtein edit distance ─────────────────────────────────────
-- Measures how many single-character changes separate two strings.
-- 0 = identical. Used for fuzzy "closest match" when player is wrong.
local function editDistance(a, b)
    local la, lb = #a, #b
    if la == 0 then return lb end
    if lb == 0 then return la end

    -- Wagner-Fischer algorithm
    local row = {}
    for i = 0, la do row[i] = i end

    for j = 1, lb do
        local prev = j
        for i = 1, la do
            local cost = a:sub(i,i) == b:sub(j,j) and 0 or 1
            local val  = math.min(row[i] + 1, prev + 1, row[i-1] + cost)
            row[i-1]   = prev
            prev        = val
        end
        row[la] = prev
    end
    return row[la]
end

-- Similarity score 0 → 1 (1 = identical)
local function similarity(a, b)
    local maxLen = math.max(#a, #b, 1)
    return 1 - editDistance(a, b) / maxLen
end

-- ── Main check function ───────────────────────────────────────────
-- Returns: matchedEquation, closestEquation, score
--
-- If correct:  matchedEq ≠ nil, closestEq = nil, score = 1.0
-- If wrong:    matchedEq = nil, closestEq = nearest, score 0-1
-- The closestEq is shown in the tear animation so the player
-- can see what they should have typed.

function EquationEngine.CheckEquation(input, unlockedIds)
    if not input or #input == 0 then return nil, nil, 0 end

    local normInput    = normalize(input)
    local compactInput = compact(input)

    local bestEq    = nil
    local bestScore = 0

    for _, eq in ipairs(EquationDatabase) do
        -- Skip equations the player hasn't unlocked yet
        if unlockedIds and #unlockedIds > 0 and not table.find(unlockedIds, eq.id) then
            continue
        end

        -- Pass 1: exact normalised match (fastest, checked first)
        for _, fmt in ipairs(eq.acceptedFormats or {}) do
            if normalize(fmt) == normInput then
                return eq, nil, 1.0   -- found — return immediately
            end
        end

        -- Pass 2: compact (no spaces) match for sloppy typers
        for _, fmt in ipairs(eq.acceptedFormats or {}) do
            if compact(fmt) == compactInput then
                return eq, nil, 1.0
            end
        end

        -- Pass 3: fuzzy — track the closest for the tear animation
        local topScore = 0
        for _, fmt in ipairs(eq.acceptedFormats or {}) do
            local sc = similarity(normalize(fmt), normInput)
            if sc > topScore then topScore = sc end
        end
        if topScore > bestScore then
            bestScore = topScore
            bestEq    = eq
        end
    end

    -- No exact match — return nil + closest
    return nil, bestEq, bestScore
end

-- ── Utility functions ─────────────────────────────────────────────

function EquationEngine.GetById(id)
    for _, eq in ipairs(EquationDatabase) do
        if eq.id == id then return eq end
    end
end

function EquationEngine.GetBySubject(subject)
    local out = {}
    for _, eq in ipairs(EquationDatabase) do
        if eq.tagSubject == subject then out[#out+1] = eq end
    end
    return out
end

function EquationEngine.GetUnlocked(unlockedIds)
    local out = {}
    for _, eq in ipairs(EquationDatabase) do
        if table.find(unlockedIds or {}, eq.id) then
            out[#out+1] = eq
        end
    end
    return out
end

function EquationEngine.GetAll()
    return EquationDatabase
end

return EquationEngine

-- DAMAGE FORMULA:
-- final = base × tier × spin × stability_roll × speed_bonus
--
-- base          equation.power (1-100)
-- tier          Common×1.0 | Rare×1.35 | Epic×1.75 | Legendary×2.5
-- spin          subject match ×1.25, sub-subject also ×1.15 (stack)
-- stability     100=no variance, 0=±30% variance around the mean
-- speed         answered instantly=×1.10, answered at time limit=×1.00
--
-- ═══════════════════════════════════════════════════════════════

local DamageCalc = {}

local TIER_MULT = {
    Common    = 1.00,
    Rare      = 1.35,
    Epic      = 1.75,
    Legendary = 2.50,
}

local SpinSystem = require(script.Parent.SpinSystem)

-- Stability: produces a multiplier around 1.0 with variance based on stability stat
-- stability=100 → always exactly 1.0 (predictable)
-- stability=50  → ±15% variance
-- stability=0   → ±30% variance (high risk)
local function stabilityRoll(stability)
    local variance = (1 - math.clamp(stability, 0, 100) / 100) * 0.30
    local roll     = (math.random() * 2 - 1) * variance  -- -variance to +variance
    return 1.0 + roll
end

-- Speed bonus: fast answers deal slightly more damage
-- elapsedSeconds: how long the player took to type the answer
-- turnLimit: the total time allowed per turn (default 60s)
local function speedBonus(elapsed, turnLimit)
    turnLimit = turnLimit or 60
    elapsed   = math.clamp(elapsed or 30, 0, turnLimit)
    -- Linear: 0s taken → ×1.10, 60s taken → ×1.00
    return 1.0 + (1 - elapsed / turnLimit) * 0.10
end

-- ── Main calculation ──────────────────────────────────────────────
function DamageCalc.Calculate(equation, playerData, elapsedSeconds)
    -- Validate equation fields
    local base     = tonumber(equation.power)    or 50
    local stab     = tonumber(equation.stability) or 50
    local tierName = tostring(equation.tier or "Common")

    local tierMult  = TIER_MULT[tierName] or 1.0
    local spinMult  = SpinSystem.GetMultiplier(playerData or {}, equation)
    local stabMult  = stabilityRoll(stab)
    local spdMult   = speedBonus(elapsedSeconds, 60)

    local raw   = base * tierMult * spinMult * stabMult * spdMult
    local final = math.max(1, math.round(raw))

    -- Breakdown for UI (so player sees why they got that damage)
    local breakdown = {
        base      = base,
        tier      = string.format("×%.2f  [%s]", tierMult, tierName),
        spin      = string.format("×%.2f", spinMult),
        stability = string.format("×%.2f", stabMult),
        speed     = string.format("×%.2f", spdMult),
        final     = final,
    }

    return final, breakdown
end

-- Stat card previews (shown in the Equations tab)
function DamageCalc.MaxDamage(equation)
    return math.max(1, math.round(
        (equation.power or 50)
        * (TIER_MULT[equation.tier] or 1)
        * 1.25 * 1.15   -- full spin match
        * 1.30          -- best stability
        * 1.10          -- instant answer
    ))
end

function DamageCalc.MinDamage(equation)
    return math.max(1, math.round(
        (equation.power or 50)
        * (TIER_MULT[equation.tier] or 1)
        * 0.70          -- worst stability
    ))
end

return DamageCalc

-- ── Advance turn ──────────────────────────────────────────────────
function BattleManager._advanceTurn(battleId)
    local battle = battles[battleId]
    if not battle then return end

    battle.turnNumber  += 1
    battle.turnStart    = os.time()
    battle.state        = STATES.WAITING

    -- Tick down cooldowns
    for uid, cds in pairs(battle.cooldowns) do
        for eqId, expiresAtTurn in pairs(cds) do
            if battle.turnNumber > expiresAtTurn then
                cds[eqId] = nil
            end
        end
    end

    -- Swap turn
    local p1, p2 = battle.players[1], battle.players[2]
    battle.turn = battle.turn == p1.UserId and p2.UserId or p1.UserId

    for _, plr in ipairs(battle.players) do
        fireC("BattleUpdate", plr, "NEXT_TURN", clientSnapshot(battle, plr))
    end

    BattleManager._startTimer(battleId, battle.turnNumber)
end

-- ── Submit equation ───────────────────────────────────────────────
function BattleManager.SubmitEquation(battleId, player, input)
    local battle = battles[battleId]
    if not battle                             then return end
    if battle.state ~= STATES.WAITING        then return end
    if battle.turn  ~= player.UserId         then return end

    battle.state = STATES.PROCESSING

    local elapsed = os.time() - battle.turnStart
    local pData   = DataManager.Get(player) or {}
    local unlocked= pData.unlockedEqs or {}

    -- Check equation
    local eq, closest, score = EquationEngine.CheckEquation(input, unlocked)

    -- If matched, check cooldown
    if eq and battle.cooldowns[player.UserId][eq.id] then
        closest = eq
        eq      = nil
    end

    local opponent = getOpponent(battle, player.UserId)

    if eq then
        -- ✅ CORRECT
        local damage, breakdown = DamageCalc.Calculate(eq, pData, elapsed)
        battle.health[opponent.UserId] = math.max(0, battle.health[opponent.UserId] - damage)

        if eq.cooldown and eq.cooldown > 0 then
            battle.cooldowns[player.UserId][eq.id] = battle.turnNumber + eq.cooldown
        end

        battle.log[#battle.log+1] = {
            turn=battle.turnNumber, player=player.Name,
            eq=eq.name, damage=damage, correct=true,
        }

        local payload = { eq=eq, damage=damage, breakdown=breakdown, health=battle.health }
        fireC("BattleUpdate", player,   "CARD_CORRECT", payload)
        fireC("BattleUpdate", opponent, "CARD_CORRECT", payload)

    else
        -- ❌ WRONG
        battle.log[#battle.log+1] = {
            turn=battle.turnNumber, player=player.Name, correct=false,
        }
        local payload = { closest=closest, score=score, health=battle.health }
        fireC("BattleUpdate", player,   "CARD_WRONG", payload)
        fireC("BattleUpdate", opponent, "CARD_WRONG", payload)
    end

    -- Win condition
    if battle.health[opponent.UserId] <= 0 then
        BattleManager._endBattle(battleId, player, opponent)
        return
    end

    BattleManager._advanceTurn(battleId)
end

-- ── End battle ────────────────────────────────────────────────────
function BattleManager._endBattle(battleId, winner, loser)
    local battle = battles[battleId]
    if not battle or battle.state == STATES.OVER then return end
    battle.state = STATES.OVER

    -- Notify clients
    local payload = { log=battle.log, coinsEarned=0 }

    if winner then
        payload.coinsEarned = WIN_COINS
        fireC("BattleUpdate", winner, "BATTLE_WIN",  payload)
        local wd = DataManager.Get(winner)
        if wd then
            wd.wins             = (wd.wins or 0) + 1
            wd.totalBattles     = (wd.totalBattles or 0) + 1
            wd.currentWinStreak = (wd.currentWinStreak or 0) + 1
            if wd.currentWinStreak > (wd.longestWinStreak or 0) then
                wd.longestWinStreak = wd.currentWinStreak
            end
        end
        DataManager.AddCoins(winner, WIN_COINS, "🏆 +" .. WIN_COINS .. " coins — Victory!")
    end

    if loser and loser.Parent then
        payload.coinsEarned = LOSS_COINS
        fireC("BattleUpdate", loser, "BATTLE_LOSS", payload)
        local ld = DataManager.Get(loser)
        if ld then
            ld.losses           = (ld.losses or 0) + 1
            ld.totalBattles     = (ld.totalBattles or 0) + 1
            ld.currentWinStreak = 0
        end
        DataManager.AddCoins(loser, LOSS_COINS, "+" .. LOSS_COINS .. " coins — Good fight")
    end

    -- Cleanup
    for _, plr in ipairs(battle.players) do
        playerBattle[plr.UserId] = nil
    end
    task.delay(30, function() battles[battleId] = nil end)

    print("[Battle] Ended:", battleId, "| Winner:", winner and winner.Name or "none")
end

end)

-- Each client renders their own leaves — no server involvement needed

local CollectionService = game:GetService("CollectionService")
local RunService        = game:GetService("RunService")
local Players           = game:GetService("Players")

local MAX_LEAVES  = 60   -- lower if game feels slow
local liveCount   = 0

-- Autumn colour palette — edit to match your game's aesthetic
local COLORS = {
    Color3.fromRGB(185, 92,  18),
    Color3.fromRGB(165, 62,  10),
    Color3.fromRGB(205, 145, 28),
    Color3.fromRGB(145, 80,  14),
    Color3.fromRGB(195, 115, 38),
    Color3.fromRGB(215, 165, 20),
}

local function spawnLeaf(root)
    if liveCount >= MAX_LEAVES then return end
    liveCount += 1

    -- Create the leaf part
    local leaf      = Instance.new("Part")
    leaf.Name       = "Leaf"
    leaf.Size       = Vector3.new(
        0.18 + math.random() * 0.14,
        0.03,
        0.18 + math.random() * 0.14
    )
    leaf.Color      = COLORS[math.random(#COLORS)]
    leaf.Material   = Enum.Material.SmoothPlastic
    leaf.CastShadow = false
    leaf.CanCollide = false
    leaf.Position   = root.Position + Vector3.new(
        math.random(-7, 7),
        math.random(6, 13),
        math.random(-7, 7)
    )
    leaf.Orientation = Vector3.new(
        math.random(0,360), math.random(0,360), math.random(0,360)
    )
    leaf.Parent = workspace

    -- Drift north-east with downward fall
    local vel        = Instance.new("BodyVelocity")
    vel.MaxForce     = Vector3.new(1e4, 1e4, 1e4)
    vel.Velocity     = Vector3.new(
        3.2 + math.random() * 2.2,
        -1.0 - math.random() * 0.8,
        math.random(-1, 1) * 0.6
    )
    vel.Parent = leaf

    -- Tumble spin
    local spin              = Instance.new("BodyAngularVelocity")
    spin.MaxTorque          = Vector3.new(1e4, 1e4, 1e4)
    spin.AngularVelocity    = Vector3.new(
        math.random(-4, 4),
        math.random(-2, 2),
        math.random(-4, 4)
    )
    spin.Parent = leaf

    -- Heartbeat wobble — makes leaves flutter
    local phase = math.random() * math.pi * 2
    local wobbleConn
    wobbleConn = RunService.Heartbeat:Connect(function(dt)
        if not leaf.Parent then
            wobbleConn:Disconnect()
            return
        end
        phase += dt * (0.85 + math.random() * 0.3)
        vel.Velocity = Vector3.new(
            3.2 + math.sin(phase) * 2.0,
            vel.Velocity.Y,
            math.cos(phase * 0.6) * 1.2
        )
    end)

    -- Ground detection
    local landed = false
    leaf.Touched:Connect(function(hit)
        if landed then return end
        local isGround = hit:IsA("Terrain")
            or (hit.Anchored and hit.Size.Y < 4)
        if not isGround then return end
        landed = true
        wobbleConn:Disconnect()

        vel.Velocity            = Vector3.zero
        spin.AngularVelocity    = Vector3.zero

        -- Settle then fade out
        task.delay(0.5, function()
            for i = 1, 10 do
                task.wait(0.08)
                if leaf.Parent then leaf.Transparency = i / 10 end
            end
            if leaf.Parent then leaf:Destroy() end
            liveCount -= 1
        end)
    end)

    -- Safety TTL — destroy after 25s regardless
    task.delay(25, function()
        wobbleConn:Disconnect()
        if leaf.Parent then leaf:Destroy() end
        liveCount -= 1
    end)
end

-- Start spawning from a tagged tree
local function startTree(obj)
    local root = obj:IsA("Model") and obj.PrimaryPart
               or obj:IsA("BasePart") and obj
    if not root then
        warn("[FallingLeaves] Tree has no PrimaryPart:", obj:GetFullName())
        return
    end

    task.spawn(function()
        task.wait(math.random() * 6)   -- stagger start so all trees don't burst at once
        while true do
            spawnLeaf(root)
            task.wait(1.6 + math.random() * 4.2)
        end
    end)
end

-- Initial tagged trees
local tagged = CollectionService:GetTagged("LeafTree")
for _, obj in ipairs(tagged) do
    startTree(obj)
end

-- Handle trees added later (streaming-compatible)
CollectionService:GetInstanceAddedSignal("LeafTree"):Connect(startTree)
